es6学习笔记10/Proxy

Proxy，顾名思义就是代理，可以理解成在目标对象之前架设一层拦截，外界对该对象的访问，都必须通过这层拦截，因此提供了一种机制，可以对外界的访问进行过滤和改写。

let obj= {
    time:'2017-03-11',
    name:'net',
    _r:123
}; 
//原始对象来存取真实的数据(类似供应商)

let monitor = new Proxy(obj,{
    // 拦截代理跟读取相关的属性
    get(target,key) {
        return target[key].replace('2017','2018');
    },
    //拦截对象属性的设置
    set(target,key,value) {
        if (key==='name') {
            return target[key]=value;
        } else {
            return target[key];
        }
    },
    //判断当前对象是否拥有某个属性 拦截key in object
    has(target,key) {
        if (key==='name') {
            return target[key];
        } else {
            return false;
        }
    },
    //拦截delete 
    deleteProperty(target,key) {
        if (key.indexOf('_')>-1) {
            delete target[key];
            return true;
        }else {
            return target[key];
        }
    },
    // 拦截Object.keys,Object.getOwnPropertySymbols,Object.getOwnPropertyNames
    ownKeys(target) {
        return Object.keys(target).filter(item=>item!='time');
    }
}); 
// 通过obj新生成一个对象，这个对象是映射obj的

//用户访问的是monitor
// monitor
console.log('get',monitor.time);

monitor.time='2018';
monitor.name='hahaha';
console.log('set',monitor); //time没有变化而name发生了变化

console.log('has','name' in monitor);
console.log('has','time' in monitor);

delete monitor.time;
console.log('delete',monitor);

delete monitor._r;
console.log('delete',monitor);

console.log('ownKeys',Object.keys(monitor));
