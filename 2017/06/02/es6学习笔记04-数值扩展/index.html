es6学习笔记04/数值扩展

es6在数值处理方面也新增了一些api，取一些较为常用的纪录一下。

进制

  • 二进制表示方法：0b或0B开头＋不大于1的数字集合

  • 八进制表示方法：0o开头＋不大于7的数字集合

    {
    console.log('b',0b111110111); //0b二进制
    console.log('B',0B111110111); 
    console.log(0o767);              //0o八进制
}

判断数值是否有界

用Number的isFinite()方法

{
    console.log('15',Number.isFinite(15)); //true
    console.log('NaN',Number.isFinite(NaN)); //false
    console.log('1/0',Number.isFinite(1/0)); //false
}

判断是否是数字

用isNaN()方法

{
    console.log('NaN',Number.isNaN(NaN)); 　//true
    console.log('0',Number.isNaN(0));      //false
}

判断是否是整数

用Number的isInteger()方法

{
    console.log('25',Number.isInteger(25)); //true
    console.log('25.0',Number.isInteger(25.0)); //true
    console.log('25.1',Number.isInteger(25.1)); //false
    console.log('25',Number.isInteger('25')); //false
}

值得注意的是，25.0属于整数，‘25’不属于整数。

整数界限

{
    console.log(Number.MAX_SAFE_INTEGER); //常量，整数的最大上限
    console.log(Number.MIN_SAFE_INTEGER); //常量，整数的最大下限
    console.log('10',Number.isSafeInteger(10)); //true
    console.log('a',Number.isSafeInteger('a')); //a连数字都不是 false
}

返回浮点数的整数部分

es5有 Math.floor()/Math.ceil() 向下取整或向上取整，es6中只想取整数，用Math.trunc()方法。

{
    //判断带小数的整数部分并返回
    console.log(4.1,Math.trunc(4.1));
    console.log(4.9,Math.trunc(4.9));
}

判断数字是正数负数还是零

{
    //判断一个数是正数、负数还是零
    console.log('-5',Math.sign(-5)); //-1
    console.log('0',Math.sign(0));   //0
    console.log('5',Math.sign(5));   //1 
    console.log('5',Math.sign('5')); //自动调用Number()转型函数 1
    console.log('a',Math.sign('a')); //NaN
}

立方根

{
    //立方根
    console.log('-1',Math.cbrt(-1)); //-1
    console.log('8',Math.cbrt(8));   //2
}
